#include <iostream>
#include <stdlib.h>
#include <time.h>

#include <locale.h>
using namespace std;

int check(int N){
    if ((1<= N) and (N<= 30)){
        return N;
    }
    else{
        while (not((1<= N) and (N<= 30))){
            if (N >= 31){
                cout<<"------------------------------------"<<endl;
                cout<<"Слишком большой массив!"<<endl<<"Попробуйте ввести массив поменьше:"<<endl;
                cin>>N;
            }
            else if(N<=0){
                cout<<"------------------------------------"<<endl;
                cout<<"Вы ввели отрицательный или нуливой массив!"<<endl<<"Попробуйте ввести массив поменьше:"<<endl;
                cin>>N;
            }
                
            if ((1<= N) and (N<= 30)){
                break;
            }
        }
        return N;
    }
}

int main()
{
    setlocale(0, "");
    srand(time(NULL));
    int i, N, max_el = 0, summ_el = 0, count = 0, index = 0, i_m = 0; 
    cout << "Введите размер массива: ";
    cin >> N; 
    N = check(N);
    
    int* A = new int[N];
    for (i = 0; i < N; i++) {
        *(A + i) = rand() % 1000 - 500;
    }
    
    cout<<"------------------------------------"<<endl;
    cout<< "Исходный массив" << endl;
    for (i = 0; i < N; i++) {
        cout << *(A + i) << " ";
    }
    cout << endl;
    
    max_el = *(A);
    for (i = 0; i < N; i++) {
        if ((i+1) == N){
            break;    
        }
        else{
            i_m += 1;
            if (*(A + i+1) > max_el){
                max_el = *(A + i+1);
                index = i_m;
            }
        }    
    }
    
    for (i = 0; i < index; i++){
        summ_el += *(A + i);
    }
    
    for (i = 0; i < N; i++){
        if (*(A + i) == max_el){
            *(A + i) = summ_el;
            break;
        }
    }
    
    cout<<"------------------------------------"<<endl;
    cout <<"Итоговый массив" << endl;
    for (i = 0; i < N; i++){
       cout << *(A + i) << " ";
    }
    cout<<endl;
    
    cout<<"------------------------------------"<<endl;
    for (i = 0; i < N; i++){
        if (*(A + i) < 0){
            cout<<"Адресса отрицательных элементов массива: "<<endl;
            break;
        }
    }
    
    for (i = 0; i < N; i++){
        if (*(A + i) < 0){
            cout << (A + i)<< endl;
            count += 1;
        }
    }
    
    if (count == 0){
        cout<<"В массиве нет отрицательных элементов!";
    }
    
    delete[]A;
    A = NULL;
    return 0;
}
